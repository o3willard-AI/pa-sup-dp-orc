// ────────────────────────────────────────────────────────────────────────────────
// AC2: Raw Mode Management
// ────────────────────────────────────────────────────────────────────────────────

// RawMode represents a terminal that has been switched into raw mode.
//
// In raw mode the terminal driver passes every keystroke directly to the
// application without buffering, echoing, or processing control sequences
// (e.g., Ctrl-C does not send SIGINT). This is required for interactive TUIs.
//
// # Lifecycle
//
//	raw, err := terminal.EnterRawMode(os.Stdin.Fd())
//	if err != nil { ... }
//	defer raw.Restore()   // always restore, even on panic
//
// # Thread Safety
//
// Restore() is safe to call from multiple goroutines simultaneously and from
// signal handlers. The first call performs the actual restore; subsequent calls
// are no-ops that return nil. This is implemented with sync.Once.
type RawMode struct {
	fd       uintptr     // the descriptor that was put into raw mode
	prev     *term.State // terminal state captured before entering raw mode
	once     sync.Once   // ensures Restore body executes exactly once
	restoreErr error     // error from the single real Restore attempt
}

// EnterRawMode switches the terminal identified by fd into raw mode and returns
// a *RawMode handle whose Restore method will undo the change.
//
// The caller MUST call Restore (typically via defer) to avoid leaving the
// terminal in raw mode after the program exits, which would make the shell
// unusable.
//
// Returns an error if:
//   - fd is not a terminal (use IsTerminal to check first)
//   - the underlying MakeRaw syscall fails
func EnterRawMode(fd uintptr) (*RawMode, error) {
	// Capture current state BEFORE switching to raw mode so Restore has
	// something to go back to.  If MakeRaw fails we still have a valid
	// snapshot, but we don't expose it — the caller never entered raw mode.
	prev, err := term.GetState(int(fd))
	if err != nil {
		return nil, fmt.Errorf(\"terminal: EnterRawMode GetState fd=%d: %w\", fd, err)
	}

	// MakeRaw configures the terminal for character-at-a-time input with no
	// echo.  On Unix this calls tcsetattr(fd, TCSANOW, &raw); on Windows it
	// sets ENABLE_VIRTUAL_TERMINAL_INPUT and clears ENABLE_PROCESSED_INPUT.
	if _, err := term.MakeRaw(int(fd)); err != nil {
		return nil, fmt.Errorf(\"terminal: EnterRawMode MakeRaw fd=%d: %w\", fd, err)
	}

	return &RawMode{
		fd:   fd,
		prev: prev,
	}, nil
}

// Restore returns the terminal to the state it was in before EnterRawMode was
// called.
//
// Idempotency guarantee: Restore may be called any number of times from any
// number of goroutines. Only the first call attempts the actual restore; all
// subsequent calls return nil immediately. This makes it safe to use both
// defer and explicit cleanup without double-restore races.
//
// Returns the error from the first (and only) real restore attempt. Subsequent
// calls always return nil regardless of whether the first attempt succeeded.
func (r *RawMode) Restore() error {
	r.once.Do(func() {
		// term.Restore calls tcsetattr / SetConsoleMode to reapply the saved
		// termios / CONSOLE_MODE snapshot.
		if err := term.Restore(int(r.fd), r.prev); err != nil {
			r.restoreErr = fmt.Errorf(\"terminal: Restore fd=%d: %w\", r.fd, err)
		}
	})
	// After once.Do completes (whether this goroutine ran the body or waited
	// for another goroutine to finish), restoreErr holds the definitive result.
	// Callers after the first will see nil here.
	return r.restoreErr
}

// State returns a snapshot of the terminal settings that were active
// immediately before EnterRawMode was called.
//
// This is provided so callers can pass the pre-raw state to RestoreState
// independently of the RawMode lifecycle (e.g., to restore from a signal
// handler that has lost the *RawMode pointer).
func (r *RawMode) State() *State {
	return &State{fd: r.fd, inner: r.prev}
}
