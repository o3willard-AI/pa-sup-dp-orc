## Usage Example

```go
package main

import (
    "github.com/pairadmin/pairadmin/internal/terminal"
    "github.com/pairadmin/pairadmin/internal/terminal/tmux"
)

func main() {
    // Create tmux adapter
    tmuxConfig := tmux.DefaultConfig()
    tmuxAdapter := tmux.NewAdapter(tmuxConfig)
    tmuxAdapter.Connect(ctx)
    
    // Create detector with adapter
    config := terminal.DefaultDetectorConfig()
    detector := terminal.NewDetector(config, tmuxAdapter)
    
    // Start detection
    detector.Start()
    
    // Subscribe to events
    for event := range detector.Events() {
        fmt.Printf("Event: %s - %s\n", event.Type, event.TerminalID)
    }
    
    // Get current sessions
    sessions := detector.GetSessions()
    
    // Stop when done
    detector.Stop()
}
```
