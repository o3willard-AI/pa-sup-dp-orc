**Phase:** 2 - Terminal Integration

## Objective

Update the Terminal Detector service to automatically register and use the macOS adapter when running on macOS. This enables automatic detection of Terminal.app windows alongside tmux sessions.

---

## Requirements

### 1. Platform Detection

Add platform detection to determine when to register the macOS adapter:

```go
// Check if running on macOS
func isMacOS() bool {
    return runtime.GOOS == "darwin"
}

// Check if macOS adapter is available (macOS + accessibility enabled)
func isMacOSAdapterAvailable() bool {
    if runtime.GOOS != "darwin" {
        return false
    }
    return macos.IsAccessibilityEnabled()
}
```

### 2. Auto-Registration in Detector

Modify `NewDetector()` or add initialization function to auto-register adapters:

```go
// NewDetector creates detector with auto-registered adapters
func NewDetector(config DetectorConfig) *Detector {
    var adapters []TerminalAdapter
    
    // Auto-register tmux adapter (always available if tmux installed)
    tmuxAdapter := tmux.NewAdapter(tmux.DefaultConfig())
    adapters = append(adapters, tmuxAdapter)
    
    // Auto-register macOS adapter if available
    if isMacOSAdapterAvailable() {
        macosAdapter := macos.NewAdapter(macos.DefaultConfig())
        adapters = append(adapters, macosAdapter)
    }
    
    // Create detector with adapters
    return &Detector{
        config: config,
        adapters: adapters,
        // ... other fields
    }
}
```

### 3. Accessibility Permission Prompt

On first run on macOS, if accessibility is not enabled, provide a way to request permission:

```go
// RequestAccessibilityPermission requests macOS accessibility permission
// Call this when detector starts and accessibility is not enabled
func (d *Detector) RequestAccessibilityPermission() bool {
    if runtime.GOOS != "darwin" {
        return false
    }
    return macos.RequestAccessibilityPermission()
}
```

### 4. Adapter Status Reporting

Add method to get status of registered adapters:

```go
// GetAdapterStatus returns status of each registered adapter
func (d *Detector) GetAdapterStatus() []AdapterStatus {
    d.mu.RLock()
    defer d.mu.RUnlock()
    
    var statuses []AdapterStatus
    for _, adapter := range d.adapters {
        statuses = append(statuses, AdapterStatus{
            Name: adapter.Name(),
            Available: adapter.Available(context.Background()),
            Type: getAdapterType(adapter),
        })
    }
    return statuses
}

// AdapterStatus holds status information for an adapter
type AdapterStatus struct {
    Name string
    Available bool
    Type string // "multiplexer" or "native"
}
```

### 5. Update Existing Detector

Modify `internal/terminal/detector.go` to:
- Import macOS adapter package
- Auto-register adapters on creation
- Add permission request method
- Add adapter status reporting

```go
    switch adapter.Name() {
    case "tmux", "screen":
        return "multiplexer"
    case "macos", "linux", "windows":
        return "native"
    default:
        return "unknown"
    }
```

```go
func TestDetectorAutoRegistersTmux(t *testing.T) {
    det := NewDetector(DefaultDetectorConfig())
    statuses := det.GetAdapterStatus()
    
    // tmux should always be registered (availability checked separately)
    found := false
    for _, s := range statuses {
        if s.Name == "tmux" {
            found = true
            break
        }
    }
    if !found {
        t.Error("tmux adapter should be registered")
    }

func TestDetectorMacOSAdapter(t *testing.T) {
    det := NewDetector(DefaultDetectorConfig())
    statuses := det.GetAdapterStatus()
    
    // Check macOS adapter registration based on platform
    found := false
    for _, s := range statuses {
        if s.Name == "macos" {
            found = true
            if runtime.GOOS != "darwin" {
                t.Error("macOS adapter should not be registered on non-macOS")
            }
            break
        }
    }
    
    if runtime.GOOS == "darwin" {
        // On macOS, adapter is registered (availability is separate check)
        t.Logf("macOS adapter registered: %v", found)
    }
}
```

## Example Usage

```go
// Create detector with auto-registered adapters
det := terminal.NewDetector(terminal.DefaultDetectorConfig())

// Check adapter status
statuses := det.GetAdapterStatus()
for _, s := range statuses {
    fmt.Printf("Adapter: %s, Available: %v, Type: %s\n", 
               s.Name, s.Available, s.Type)
}

// Request permission if needed (macOS only)
if runtime.GOOS == "darwin" {
    if !terminal.IsAccessibilityEnabled() {
        fmt.Println("Requesting accessibility permission...")
        det.RequestAccessibilityPermission()
    }
}

// Start detection
det.Start()
defer det.Stop()

// Subscribe to events
for event := range det.Events() {
    fmt.Printf("Event: %s - %s\n", event.Type, event.TerminalID)
}
```
