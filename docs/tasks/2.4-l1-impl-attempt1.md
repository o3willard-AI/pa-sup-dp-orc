```go
// Package ui provides Wails backend handlers for UI features like terminals.
package ui

import (
	"bufio"
	"context"
	"fmt"
	"io"
	"os/exec"
	"sync"

	"github.com/wailsapp/wails/v2/pkg/runtime"

	"pairadmin/internal/detector" // From Task 2.3: Service detector
)

// TerminalHandlers manages multiple terminal sessions with real-time streaming to frontend.
// Integrates with Detector to start sessions for detected services.
// L0 Learning: No file I/O/JSON needed here; focus on Wails events + exec pipes for "terminal".
type TerminalHandlers struct {
	detector *detector.Detector
	sessions map[string]*TerminalSession // sessionID -> session
	mu       sync.RWMutex
}

// TerminalSession represents a single terminal session (pseudo-pty via pipes).
// Streams output via goroutines/EventsEmit for xterm.js frontend.
type TerminalSession struct {
	ID      string
	Service detector.Service
	Cmd     *exec.Cmd
	Stdin   io.WriteCloser
	// Stdout/Stderr managed by streaming goroutines
}

// NewTerminalHandlers creates and returns handlers bound for Wails.
// Pass detector from Task 2.3.
func NewTerminalHandlers(d *detector.Detector) *TerminalHandlers {
	return &TerminalHandlers{
		detector: d,
		sessions: make(map[string]*TerminalSession),
	}
}

// ListSessions returns active session IDs for frontend tabs.
// L0 Learning: Use RWMutex for concurrent-safe listing.
func (t *TerminalHandlers) ListSessions(ctx context.Context) []string {
	t.mu.RLock()
	defer t.mu.RUnlock()
	ids := make([]string, 0, len(t.sessions))
	for id := range t.sessions {
		ids = append(ids, id)
	}
	return ids
}

// StartSessionForService starts a bash shell session for a detected service.
// Emits "terminal-session-started" event; auto-streams output.
// L0 Learning: Use exec.Command pipes + goroutines for real-time EventsEmit (no polling).
// Assumes frontend xterm.js listens for "terminal-output".
func (t *TerminalHandlers) StartSessionForService(serviceName string, ctx context.Context) (string, error) {
	services, err := t.detector.Detect(ctx)
	if err != nil {
		return "", fmt.Errorf("detector failed: %w", err)
	}
	var service detector.Service
	for _, s := range services {
		if s.Name == serviceName {
			service = s
			break
		}
	}
	if service.Name == "" {
		return "", fmt.Errorf("service %q not detected", serviceName)
	}

	// Simple ID: serviceName + "-tab1" (extendable for multiples)
	sessionID := fmt.Sprintf("%s-tab1", serviceName)

	t.mu.Lock()
	if _, exists := t.sessions[sessionID]; exists {
		t.mu.Unlock()
		return "", fmt.Errorf("session %q already exists", sessionID)
	}
	t.mu.Unlock()

	// Start bash (portable shell; customize per service if needed)
	cmd := exec.Command("bash")
	stdin, err := cmd.StdinPipe()
	if err != nil {
		return "", fmt.Errorf("stdin pipe: %w", err)
	}
	stdout, err := cmd.StdoutPipe()
	if err != nil {
		return "", fmt.Errorf("stdout pipe: %w", err)
	}
	stderr, err := cmd.StderrPipe()
	if err != nil {
		return "", fmt.Errorf("stderr pipe: %w", err)
	}

	if err := cmd.Start(); err != nil {
		return "", fmt.Errorf("start cmd: %w", err)
	}

	session := &TerminalSession{
		ID:      sessionID,
		Service: service,
		Cmd:     cmd,
		Stdin:   stdin,
	}

	t.mu.Lock()
	t.sessions[sessionID] = session
	t.mu.Unlock()

	// L0 Learning: Goroutines stream lines in background; EventsEmit pushes to frontend instantly.
	go t.streamPipe(sessionID, stdout, "stdout")
	go t.streamPipe(sessionID, stderr, "stderr")

	// Notify frontend
	runtime.EventsEmit(ctx, "terminal-session-started", map[string]any{
		"id":      sessionID,
		"service": service.Name,
	})
	return sessionID, nil
}

// streamPipe reads lines from pipe and emits to frontend.
// Closes on EOF (session end).
// L0 Learning: bufio.Scanner for line-by-line; prevents blocking.
func (t *TerminalHandlers) streamPipe(sessionID string, r io.ReadCloser, streamType string) {
	defer r.Close()
	scanner := bufio.NewScanner(r)
	for scanner.Scan() {
		line := scanner.Text()
		runtime.EventsEmit(context.Background(), "terminal-output", map[string]string{
			"sessionID": sessionID,
			"stream":    streamType,
			"line":      line,
		})
	}
	if err := scanner.Err(); err != nil {
		runtime.EventsEmit(context.Background(), "terminal-error", map[string]string{
			"sessionID": sessionID,
			"error":     err.Error(),
		})
	}
}

// SendInput forwards user input to session stdin.
// L0 Learning: Append \n for shell execution; handle close.
func (t *TerminalHandlers) SendInput(sessionID, input string, ctx context.Context) error {
	t.mu.RLock()
	session, exists := t.sessions[sessionID]
	t.mu.RUnlock()
	if !exists {
		return fmt.Errorf("session %q not found", sessionID)
	}
	_, err := session.Stdin.Write([]byte(input + "\n"))
	if err != nil {
		return fmt.Errorf("write input: %w", err)
	}
	return nil
}

// SelectSession selects a session for focus; emits event for frontend highlight.
// L0 Learning: Events for UI state sync (no direct DOM manip).
func (t *TerminalHandlers) SelectSession(sessionID string, ctx context.Context) error {
	t.mu.RLock()
	_, exists := t.sessions[sessionID]
	t.mu.RUnlock()
	if !exists {
		return fmt.Errorf("session %q not found", sessionID)
	}
	runtime.EventsEmit(ctx, "terminal-session-selected", sessionID)
	return nil
}

// SwitchTab emits event for tab switching (assumes frontend tabs map to sessions).
// L0 Learning: Pure event-driven; backend stateless on tabs.
func (t *TerminalHandlers) SwitchTab(tabID string, ctx context.Context) {
	runtime.EventsEmit(ctx, "terminal-tab-switched", tabID)
}

// CloseSession kills a session and cleans up.
func (t *TerminalHandlers) CloseSession(sessionID string, ctx context.Context) error {
	t.mu.Lock()
	session, exists := t.sessions[sessionID]
	delete(t.sessions, sessionID)
	t.mu.Unlock()
	if !exists {
		return fmt.Errorf("session %q not found", sessionID)
	}
	session.Cmd.Process.Kill()
	runtime.EventsEmit(ctx, "terminal-session-closed", sessionID)
	return nil
}
```
